Validate Postgres Data with Queries

Run queries to verify data is correctly populated before setting up CDC.

Query 1: Order Summary with Customer and Product Details

SQL (PostgreSQL) 
SELECT
    o.order_id,
    o.order_date,
    c.first_name || ' ' || c.last_name AS customer_name,
    p.product_name,
    p.price,
    o.quantity,
    (p.price * o.quantity) AS total_cost,
    o.status
FROM ecommerce.orders o
JOIN ecommerce.customers c ON o.customer_id = c.customer_id
JOIN ecommerce.products p ON o.product_id = p.product_id
ORDER BY o.order_date;

Query 2: Top Customers by Spending

SQL (PostgreSQL) 
SELECT
    c.first_name,
    c.last_name,
    COUNT(o.order_id) AS number_of_orders,
    SUM(p.price * o.quantity) AS total_spent
FROM ecommerce.customers c
JOIN ecommerce.orders o ON c.customer_id = o.customer_id
JOIN ecommerce.products p ON o.product_id = p.product_id
GROUP BY c.first_name, c.last_name
ORDER BY total_spent DESC
LIMIT 5;

Query 3: Revenue by Category

SQL (PostgreSQL) 
SELECT
    p.category,
    SUM(o.quantity) AS items_sold,
    SUM(p.price * o.quantity) AS revenue
FROM ecommerce.orders o
JOIN ecommerce.products p ON o.product_id = p.product_id
GROUP BY p.category
ORDER BY revenue DESC;
